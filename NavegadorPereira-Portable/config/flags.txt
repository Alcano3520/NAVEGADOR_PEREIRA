# Flags de Chromium para privacidad, rendimiento y tema vintage
# Estos se aplican automaticamente al ejecutar Pereira.bat

# PRIVACIDAD
--disable-background-networking
--disable-sync
--disable-features=MediaRouter
--disable-default-apps
--disable-extensions-except
--no-first-run
--no-default-browser-check
--disable-external-intent-requests
--bwsi
--disable-logging
--disable-breakpad

# DNS Y RED
--disable-domain-reliability
--disable-features=NetworkPrediction
--dns-prefetch-disable

# TRACKING
--enable-features=EnableTLS13EarlyData
--force-enable-metrics-reporting=0
--enable-do-not-track

# SUPRIMIR MENSAJE API KEYS
--simulate-outdated-no-au="Tue, 31 Dec 2099 23:59:59 GMT"

# TEMA OSCURO VINTAGE
--force-dark-mode
--enable-features=WebUIDarkMode
--enable-features=WebContentsForceDark
--enable-features=CSSColorSchemeUARendering

# RENDIMIENTO
--enable-features=ParallelDownloading
--enable-gpu-rasterization
--enable-zero-copy

# SEGURIDAD
--ssl-version-min=tls1.2
--enable-features=StrictOriginIsolation

# UI/UX
--disable-popup-blocking=false
--enable-smooth-scrolling

# EXPERIMENTAL (comenta si causan problemas)
--enable-quic
